                    return False

                    name='LiveDataSnapshotScheduler',
                    daemon=True,

                    raise exc

                log_event(
                    'live_data_snapshot_restored',
                    {
                        'snapshot_name': snapshot_name,
                        'restored_to': self.source_dir,
                        'backup_path': backup_dir,
                    },
                )
                return {
                    'snapshot': snapshot_name,
                    'restored_to': self.source_dir,
                    'backup_path': backup_dir,

        def _run_scheduler(self) -> None:
            while not self._stop_event.is_set():
                if self.interval_seconds <= 0:
                    break
                if self._stop_event.wait(self.interval_seconds):
                    break

        def _create_snapshot(self, trigger: str) -> Dict[str, object]:

                        'snapshot_root': self.snapshot_root,

                    'duration_seconds': duration,
                    'trigger': trigger,
                }

        def _generate_snapshot_suffix(self) -> str:
            base = datetime.now().strftime('%Y%m%d_%H%M%S')
            suffix = base
            counter = 0
            while True:
                tmp_candidate = os.path.join(self.snapshot_root, f'{self._TMP_PREFIX}{suffix}')
                final_candidate = os.path.join(self.snapshot_root, f'{self._SNAPSHOT_PREFIX}{suffix}')
                if not os.path.exists(tmp_candidate) and not os.path.exists(final_candidate):
                    return suffix
                counter += 1
                suffix = f'{base}_{counter:02d}'

        def _enforce_retention(self) -> None:
            if self.retention_limit <= 0:

            if len(entries) <= self.retention_limit:
                return

                shutil.rmtree(entry['path'], ignore_errors=True)
                log_event(
                    'live_data_snapshot_pruned',
                    {
                        'snapshot_name': entry['name'],
                        'snapshot_root': self.snapshot_root,
                    },
                )

            records: List[Dict[str, object]] = []

                path = os.path.join(self.snapshot_root, name)
                if not os.path.isdir(path):
                    continue
                timestamp = self._extract_timestamp(name)
                if not timestamp:
                    continue
                records.append({'name': name, 'path': path, 'timestamp': timestamp})
            records.sort(key=lambda item: item['timestamp'], reverse=True)
            return records

        @classmethod
        def _extract_timestamp(cls, name: str) -> Optional[datetime]:
            prefix_len = len(cls._SNAPSHOT_PREFIX)
            if not name.startswith(cls._SNAPSHOT_PREFIX) or len(name) <= prefix_len:

            raw = name[prefix_len:]
            base = raw[:15]
            try:
                return datetime.strptime(base, '%Y%m%d_%H%M%S')
            except ValueError:
                return None
