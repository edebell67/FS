## 1. Understanding of Requirements

Add columns to the Strategy Performance table (`strategy_performance.html`):

1. **"#Trades"** - New column showing total trade count (#buy + #sell), placed next to "Total Net"
2. **"Auto Buy"** - Toggle button to add/remove `{strategy}_{parm}_buy_net` activations
3. **"Auto Sell"** - Toggle button to add/remove `{strategy}_{parm}_sell_net` activations

The Auto Buy/Sell buttons allow users to add or remove the selected row's strategy/product combination to/from the `activations.json` file.

### Expected `activations.json` Structure
```json
{
    "breakout_R_Rev_2_tp10.0_sl10.0_sell_net": {
        "active": true,
        "manual": true,
        "activated_at": "2025-12-30T13:38:25.911072",
        "products": ["EUR"]
    },
    "breakout_R_Rev_3_tp10.0_sl10.0_buy_net": {
        "active": true,
        "manual": true,
        "activated_at": "2025-12-30T23:04:56.574272",
        "products": ["GBPEUR_C"]
    }
}
```

### Activation Key Formats
- **Buy Key**: `{Strategy}_{Parm}_buy_net` (e.g., `breakout_R_Rev_3_tp10.0_sl10.0_buy_net`)
- **Sell Key**: `{Strategy}_{Parm}_sell_net` (e.g., `breakout_R_Rev_3_tp10.0_sl10.0_sell_net`)
- **Products array**: Contains the product code from the row (e.g., `["GBPEUR_C"]`)

## 4. Implementation Details

### 4.1 Activation Key Construction
```javascript
// Build the activation key from row data
// direction = 'buy' or 'sell'
function buildActivationKey(strategy, parm, direction) {
    // strategy = "breakout_R_Rev_3"
    // parm = "2_0.0001_10.0_10.0" (raw parm without braces)
    // direction = "buy" or "sell"
    return `${strategy}_${parm}_${direction}_net`;
}
```

### 4.3 Toggle API Call
```javascript
async function toggleAutoActivation(strategy, parm, product, direction, button) {
    const key = buildActivationKey(strategy, parm, direction);
    const isCurrentlyActive = activations[key]?.products?.includes(product.toUpperCase());
    
    const payload = {
        [key]: {
            active: !isCurrentlyActive,
            manual: true,
            products: [product]
        }
    };
    
    const response = await fetch('http://localhost:5000/api/activations', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
    });
    
    if (response.ok) {
        const data = await response.json();
        activations = data.activations;
        // Update button visual state
        const newState = !isCurrentlyActive;
        button.classList.toggle('active', newState);
        button.textContent = newState ? 'ON' : 'OFF';
    }
}
```

### 4.4 Check Active State
```javascript
function isAutoActive(strategy, parm, product, direction) {
    const key = buildActivationKey(strategy, parm, direction);
    const entry = activations[key];
    if (!entry || !entry.active) return false;
    return entry.products?.includes(product.toUpperCase()) || false;
}
```
